#!/usr/bin/env python3
"""Smart CD Faster - Database module."""

import os
import sqlite3
from config import DATA_DIR, DB_PATH

class Database:
    def __init__(self, db_path=None):
        self.db_path = db_path or DB_PATH
        self._conn = None
        self._ensure_tables()

    def _ensure_tables(self):
        os.makedirs(DATA_DIR, exist_ok=True)
        conn = sqlite3.connect(self.db_path)
        c = conn.cursor()
        c.execute("""CREATE TABLE IF NOT EXISTS directory_visits (
            id INTEGER PRIMARY KEY AUTOINCREMENT, path TEXT NOT NULL,
            timestamp REAL NOT NULL, weekday INTEGER NOT NULL, hour INTEGER NOT NULL,
            session_id TEXT, prev_dir TEXT, dwell_time REAL DEFAULT 0)""")
        c.execute("CREATE INDEX IF NOT EXISTS idx_visits_path ON directory_visits(path)")
        c.execute("CREATE INDEX IF NOT EXISTS idx_visits_timestamp ON directory_visits(timestamp)")
        c.execute("CREATE INDEX IF NOT EXISTS idx_visits_prev ON directory_visits(prev_dir, path)")

        c.execute("""CREATE TABLE IF NOT EXISTS recommendation_sessions (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            timestamp REAL NOT NULL, current_dir TEXT, prev_dir TEXT,
            session_uuid TEXT UNIQUE)""")
        c.execute("""CREATE TABLE IF NOT EXISTS recommendation_items (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            session_id INTEGER NOT NULL, letter TEXT NOT NULL,
            path TEXT NOT NULL, rank INTEGER NOT NULL, was_selected INTEGER DEFAULT 0,
            f_freq REAL DEFAULT 0, f_rec REAL DEFAULT 0, f_time REAL DEFAULT 0,
            f_trans REAL DEFAULT 0, f_dwell REAL DEFAULT 0,
            FOREIGN KEY (session_id) REFERENCES recommendation_sessions(id))""")
        conn.commit()
        conn.close()

    @property
    def conn(self):
        if self._conn is None:
            self._conn = sqlite3.connect(self.db_path, timeout=30.0)
            self._conn.row_factory = sqlite3.Row
            self._conn.execute("PRAGMA journal_mode=WAL")
        return self._conn

    def record_visit(self, path, timestamp, weekday, hour, session_id=None, prev_dir=None, dwell_time=0):
        self.conn.cursor().execute("INSERT INTO directory_visits (path, timestamp, weekday, hour, session_id, prev_dir, dwell_time) VALUES (?, ?, ?, ?, ?, ?, ?)",
            (path, timestamp, weekday, hour, session_id, prev_dir, dwell_time))
        self.conn.commit()

    def get_prediction_data(self, limit=60):
        c = self.conn.cursor()
        c.execute("SELECT path, COUNT(*) as visit_count, MAX(timestamp) as last_visit, SUM(dwell_time) as total_dwell_time FROM directory_visits GROUP BY path ORDER BY last_visit DESC LIMIT ?", (limit,))
        candidates, paths, max_visits, max_dwell = [], [], 1, 1
        for r in c.fetchall():
            d = dict(r)
            candidates.append(d); paths.append(d['path'])
            max_visits = max(max_visits, d['visit_count'])
            max_dwell = max(max_dwell, d['total_dwell_time'])
        hour_dists = self.get_all_hour_distributions(paths)
        for cand in candidates:
            cand['hour_distribution'] = hour_dists.get(cand['path'], [])
        return candidates, max_visits, max_dwell

    def get_all_hour_distributions(self, paths):
        if not paths: return {}
        c = self.conn.cursor()
        c.execute(f"SELECT path, hour, COUNT(*) as count FROM directory_visits WHERE path IN ({','.join('?'*len(paths))}) GROUP BY path, hour", paths)
        result = {}
        for r in c.fetchall(): result.setdefault(r['path'], []).append((r['hour'], r['count']))
        return result

    def get_transitions_from(self, prev_dir):
        rows = self.conn.cursor().execute("SELECT path, COUNT(*) as count FROM directory_visits WHERE prev_dir = ? GROUP BY path", (prev_dir,)).fetchall()
        total = sum(r['count'] for r in rows)
        return {r['path']: r['count']/total for r in rows} if total > 0 else {}

    def get_second_order_transitions(self, prev_dir, current_dir):
        if not prev_dir: return self.get_transitions_from(current_dir)
        query = "SELECT v2.path, COUNT(*) as cnt FROM directory_visits v1 JOIN directory_visits v2 ON v1.session_id = v2.session_id AND v2.timestamp > v1.timestamp AND v2.prev_dir = v1.path WHERE v1.prev_dir = ? AND v1.path = ? GROUP BY v2.path"
        rows = self.conn.cursor().execute(query, (prev_dir, current_dir)).fetchall()
        total = sum(r['cnt'] for r in rows)
        if total > 0: return {r['path']: r['cnt'] / total for r in rows}
        return self.get_transitions_from(current_dir)

    def create_recommendation_session(self, current_dir, prev_dir, session_uuid):
        import time
        c = self.conn.cursor()
        c.execute("INSERT OR IGNORE INTO recommendation_sessions (timestamp, current_dir, prev_dir, session_uuid) VALUES (?, ?, ?, ?)", (time.time(), current_dir, prev_dir, session_uuid))
        self.conn.commit()
        row = c.execute("SELECT id FROM recommendation_sessions WHERE session_uuid = ?", (session_uuid,)).fetchone()
        return row['id'] if row else None

    def add_recommendation_item(self, session_id, letter, path, rank, features):
        self.conn.cursor().execute("INSERT INTO recommendation_items (session_id, letter, path, rank, f_freq, f_rec, f_time, f_trans, f_dwell) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)",
            (session_id, letter, path, rank, features.get('f_freq',0), features.get('f_rec',0), features.get('f_time',0), features.get('f_trans',0), features.get('f_dwell',0)))

    def commit(self): self.conn.commit()

    def update_dwell_time(self, path, dwell_time):
        """更新指定路径最近一条记录的停留时间（单语句避免TOCTOU）"""
        c = self.conn.cursor()
        # 单语句：先查后更，避免竞态条件
        c.execute('''UPDATE directory_visits SET dwell_time = ?
                     WHERE id = (SELECT id FROM directory_visits
                                 WHERE path = ? ORDER BY timestamp DESC LIMIT 1)''',
                  (dwell_time, path))
        self.conn.commit()

    def mark_selected(self, session_uuid, letter):
        c = self.conn.cursor()
        row = c.execute("SELECT id FROM recommendation_sessions WHERE session_uuid = ?", (session_uuid,)).fetchone()
        if row:
            c.execute("UPDATE recommendation_items SET was_selected = 1 WHERE session_id = ? AND letter = ?", (row['id'], letter))
            self.conn.commit()

    def get_session_and_items(self, session_uuid):
        c = self.conn.cursor()
        session_row = c.execute("SELECT * FROM recommendation_sessions WHERE session_uuid = ?", (session_uuid,)).fetchone()
        if not session_row: return None, []
        items = [dict(r) for r in c.execute("SELECT * FROM recommendation_items WHERE session_id = ? ORDER BY rank", (session_row['id'],)).fetchall()]
        return dict(session_row), items

    def get_selection_count(self):
        row = self.conn.cursor().execute("SELECT COUNT(*) as cnt FROM recommendation_items WHERE was_selected = 1").fetchone()
        return row['cnt'] if row else 0

    def get_learning_stats(self):
        c = self.conn.cursor()
        c.execute("SELECT (SELECT COUNT(*) FROM recommendation_items WHERE was_selected = 1) as total_selections, (SELECT COUNT(*) FROM recommendation_sessions) as total_sessions")
        row = c.fetchone()
        return {'total_selections': row['total_selections'] or 0, 'total_sessions': row['total_sessions'] or 0}

    def close(self):
        if self._conn: self._conn.close(); self._conn = None

    def __enter__(self):
        return self

    def __exit__(self, *_):
        self.close()
