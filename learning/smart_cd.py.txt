#!/usr/bin/env python3
"""Smart CD Faster - 统一入口脚本

用法:
    smart_cd.py --gen <dir> [prev_dir]    # 后台生成推荐
    smart_cd.py --show                    # 显示推荐
    smart_cd.py --status                  # 显示学习状态和权重
    smart_cd.py --record-visit <path> <prev_dir> <seconds> # 记录完整访问（停留>=阈值）
    smart_cd.py --letter <dir> <letter> [prev_dir] # 获取字母路径
    smart_cd.py --select <letter>         # 记录选择

注意: 停留时间不足 MIN_DWELL_SECONDS(默认5秒) 的目录不会被记录
"""

import os
import sys
import time
import json
from datetime import datetime

# 添加模块路径
sys.path.insert(0, os.path.dirname(os.path.abspath(__file__)))

from config import get_display_config, get_color_code, is_excluded, is_recommendation_enabled, CACHE_FILE, normalize_path
from database import Database
from predictor import Predictor, get_cached_recommendations
from learner import Learner

HOME = os.path.expanduser('~')
SESSION_ID = os.environ.get('SMART_CD_SESSION', '')
MIN_DWELL_SECONDS = int(os.environ.get('MIN_DWELL_SECONDS', '5'))  # 最小停留阈值


# ===== --gen: 后台生成推荐 =====
def cmd_gen():
    if len(sys.argv) < 3:
        print("Usage: smart_cd.py --gen <current_dir> [prev_dir]", file=sys.stderr)
        sys.exit(1)

    current_dir = sys.argv[2]
    prev_dir = sys.argv[3] if len(sys.argv) > 3 else None

    with Predictor() as predictor:
        predictor.generate_recommendations(current_dir, prev_dir)


# ===== --show: 显示推荐 =====
def get_terminal_width():
    try:
        return os.get_terminal_size().columns
    except:
        return 80

def cmd_show():
    if not is_recommendation_enabled():
        print("\033[0;90m(智能推荐已禁用，设置 recommendation_limit > 0 启用)\033[0m")
        return

    cache = get_cached_recommendations()
    if not cache or not cache.get('recommendations'):
        print("\033[0;90m(无推荐数据，请先执行 ll)\033[0m")
        return

    display = get_display_config()
    letter_clr = get_color_code(display['letter_color'])
    path_clr = get_color_code(display['path_color'])
    reset_clr = get_color_code('default')
    gray_clr = get_color_code('gray')
    cyan_clr = get_color_code('cyan')

    w = get_terminal_width()

    # 标题行
    header = "智能推荐:"
    header_display_width = len(header) + len([c for c in header if ord(c) > 127])
    left_len = (w - header_display_width) // 2
    right_len = w - header_display_width - left_len
    print(gray_clr + '-' * left_len + cyan_clr + header + reset_clr + gray_clr + '-' * right_len + reset_clr)

    shown = 0
    limit = display['limit']

    for rec in cache['recommendations']:
        if shown >= limit:
            break

        path = rec['path']
        letter = rec['letter']

        if is_excluded(path):
            continue

        if not os.path.isdir(path):
            continue

        if path.startswith(HOME):
            display_path = '~' + path[len(HOME):]
        else:
            display_path = path

        letter_display = display['letter_format'].replace('a', letter)

        score = min(int(rec.get('score', 0) * 100), 99)
        visits = min(rec.get('visit_count', 0), 99)

        stats = f"S{score:02d} V{visits:02d}"
        line = f"{gray_clr}{stats}{reset_clr} {letter_clr}{letter_display}{reset_clr} {path_clr}{display_path}{reset_clr}"

        if len(display_path) > w - 20:
            display_path = display_path[:w - 23] + '...'
            line = f"{gray_clr}{stats}{reset_clr} {letter_clr}{letter_display}{reset_clr} {path_clr}{display_path}{reset_clr}"

        print(line)
        shown += 1

    if shown == 0:
        print(f"{gray_clr}(所有推荐已被排除或不存在){reset_clr}")

    print(gray_clr + '-' * w + reset_clr)

    if shown > 0:
        first_letter = cache['recommendations'][0]['letter']
        letter_display = display['letter_format'].replace('a', first_letter)
        print(f"{gray_clr}提示: cd {letter_display} 快速进入{reset_clr}")


# ===== --status: 显示学习状态和权重 =====
def cmd_status():
    """显示学习系统状态，包括权重配置、触发条件、选择统计"""
    cyan = '\033[1;36m'
    green = '\033[0;32m'
    yellow = '\033[1;33m'
    gray = '\033[0;90m'
    reset = '\033[0m'

    with Learner() as learner:
        status = learner.get_status()

    weights = status['current_weights']
    config = status['config']
    selections = status['selections_count']

    # 默认权重对比
    default = {'frequency_weight': 0.30, 'recency_weight': 0.25, 'time_pattern_weight': 0.10, 'transition_weight': 0.15, 'dwell_weight': 0.20}

    print()
    print(f"{cyan}=== 智能推荐学习状态 ==={reset}")
    print()

    # 权重对比表
    print(f"{cyan}权重配置:{reset}")
    print(f"{'因子':<20} {'默认':<8} {'当前':<8} {'变化'}")
    print(gray + '-' * 45 + reset)
    for name in ['frequency_weight', 'recency_weight', 'time_pattern_weight', 'transition_weight', 'dwell_weight']:
        def_val = default.get(name, 0)
        cur_val = weights.get(name, def_val)
        change = cur_val - def_val
        sign = '+' if change >= 0 else ''
        # 简化显示名称
        display_name = name.replace('_weight', '').replace('_', ' ').title()
        print(f"{display_name:<20} {def_val:<8.2f} {cur_val:<8.4f} {sign}{change:.4f}")

    print()

    # 学习触发条件
    print(f"{cyan}学习触发条件:{reset}")
    print(f"  排名阈值:     > {config['rank_threshold']} (选择 d 之后才触发)")
    print(f"  评分差距:     > {config['score_gap_threshold']} (与首选差距足够大)")
    print(f"  最少选择:     ≥ 10 次")
    print(f"  更新间隔:     每 5 次")
    print(f"  单因子变化:   ≤ {config['max_change']} (≤ 1%)")

    print()

    # 统计信息
    print(f"{cyan}统计信息:{reset}")
    print(f"  总选择次数:   {selections}")
    print(f"  学习触发:     {max(0, (selections - 10) // 5)} 次")

    print()

    # 数据库统计
    with Database() as db:
        stats = db.get_learning_stats()
        c = db.conn.cursor()
        c.execute("SELECT COUNT(*) FROM directory_visits")
        visits = c.fetchone()[0]
        c.execute("SELECT COUNT(DISTINCT path) FROM directory_visits")
        unique_paths = c.fetchone()[0]

    print(f"{cyan}数据库:{reset}")
    print(f"  总访问记录:   {visits}")
    print(f"  独立路径:     {unique_paths}")
    print(f"  推荐session:  {stats['total_sessions']}")

    print()


# ===== --visit: 记录访问（仅记录访问，不记录停留） =====
def cmd_visit():
    if len(sys.argv) < 3:
        sys.exit(1)

    path = normalize_path(sys.argv[2])
    prev_dir = sys.argv[3] if len(sys.argv) > 3 else None
    if prev_dir:
        prev_dir = normalize_path(prev_dir)

    if not os.path.isdir(path):
        sys.exit(0)

    now = datetime.now()
    timestamp = time.time()
    weekday = now.weekday()
    hour = now.hour

    with Database() as db:
        db.record_visit(path, timestamp, weekday, hour, SESSION_ID, prev_dir)


# ===== --record-visit: 记录完整访问（包含停留时间） =====
def cmd_record_visit():
    """离开目录时记录完整访问信息（仅停留>=阈值的目录）
    用法: --record-visit <path> <prev_dir> <dwell_seconds>
    """
    if len(sys.argv) < 5:
        print("Usage: smart_cd.py --record-visit <path> <prev_dir> <dwell_seconds>", file=sys.stderr)
        sys.exit(1)

    path = normalize_path(sys.argv[2])
    prev_dir = normalize_path(sys.argv[3]) if sys.argv[3] else None
    dwell_time = float(sys.argv[4])

    # 停留时间不足阈值，不记录
    if dwell_time < MIN_DWELL_SECONDS:
        sys.exit(0)

    if not os.path.isdir(path):
        sys.exit(0)

    # 使用离开时的时间戳计算访问时间
    now = datetime.now()
    timestamp = time.time() - dwell_time  # 访问时间 = 当前时间 - 停留时长
    weekday = datetime.fromtimestamp(timestamp).weekday()
    hour = datetime.fromtimestamp(timestamp).hour

    with Database() as db:
        db.record_visit(path, timestamp, weekday, hour, SESSION_ID, prev_dir, dwell_time)


# ===== --letter: 获取字母路径 =====
def cmd_letter():
    if len(sys.argv) < 4:
        sys.exit(1)

    current_dir = sys.argv[2]
    letter = sys.argv[3].lower()
    prev_dir = sys.argv[4] if len(sys.argv) > 4 else None

    # 先尝试从缓存获取
    cache = get_cached_recommendations()
    if cache and cache.get('current_dir') == current_dir:
        for rec in cache.get('recommendations', []):
            if rec['letter'] == letter:
                print(rec['path'])
                return

    # 缓存无效则重新计算
    with Predictor() as predictor:
        path = predictor.get_path_by_letter(current_dir, letter, prev_dir)
        if path:
            print(path)


# ===== --select: 记录选择 =====
def cmd_select():
    if len(sys.argv) < 3:
        sys.exit(1)

    letter = sys.argv[2].lower()

    cache = get_cached_recommendations()
    if not cache:
        sys.exit(1)

    session_uuid = cache.get('session_uuid')
    if not session_uuid:
        sys.exit(1)

    with Learner() as learner:
        learner.record_selection(session_uuid, letter)


# ===== --update-dwell: 更新停留时间 =====
def cmd_update_dwell():
    """更新指定路径的停留时间（Shell 变量触发）
    仅记录停留时间 >= MIN_DWELL_SECONDS 的目录（过滤短暂路过）
    """
    if len(sys.argv) < 4:
        print("Usage: smart_cd.py --update-dwell <path> <seconds>", file=sys.stderr)
        sys.exit(1)

    path = normalize_path(sys.argv[2])
    dwell_time = float(sys.argv[3])

    # 停留时间不足阈值，不记录（过滤短暂路过）
    if dwell_time < MIN_DWELL_SECONDS:
        sys.exit(0)

    if not os.path.isdir(path):
        sys.exit(0)

    with Database() as db:
        db.update_dwell_time(path, dwell_time)


# ===== 主入口 =====
def main():
    if len(sys.argv) < 2:
        print(__doc__)
        sys.exit(1)

    cmd = sys.argv[1]

    if cmd == '--gen':
        cmd_gen()
    elif cmd == '--show':
        cmd_show()
    elif cmd == '--status':
        cmd_status()
    elif cmd == '--visit':
        cmd_visit()
    elif cmd == '--record-visit':
        cmd_record_visit()
    elif cmd == '--update-dwell':
        cmd_update_dwell()
    elif cmd == '--letter':
        cmd_letter()
    elif cmd == '--select':
        cmd_select()
    else:
        print(f"未知命令: {cmd}", file=sys.stderr)
        print(__doc__)
        sys.exit(1)


if __name__ == '__main__':
    main()
