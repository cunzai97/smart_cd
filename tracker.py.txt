#!/usr/bin/env python3
"""Smart CD - Tracker module."""

import os
import sys
import time
import uuid
from datetime import datetime
from config import FILE_ACCESS_ERRORS
from database import Database

class Tracker:
    def __init__(self, db=None):
        self.db = db or Database()
        self._session_id = None

    def _get_session_id(self):
        if self._session_id is None:
            self._session_id = os.environ.get('SMART_CD_SESSION', '') or str(uuid.uuid4())[:8]
        return self._session_id

    def record_visit(self, path, prev_dir=None):
        now = time.time()
        dt = datetime.fromtimestamp(now)
        abs_prev = os.path.abspath(prev_dir) if prev_dir else None
        self.db.record_visit(os.path.abspath(path), now, dt.weekday(), dt.hour, self._get_session_id(), abs_prev)

    def close(self): self.db.close()
    def __enter__(self): return self
    def __exit__(self, *_): self.close()

def get_terminal_directories():
    """Get directories of all active terminals (Linux only)."""
    dir_map = {}
    if sys.platform != 'linux': return dir_map
    try:
        for entry in os.listdir('/proc'):
            if not entry.isdigit() or int(entry) < 100: continue
            try:
                cwd = os.readlink(os.path.join('/proc', entry, 'cwd'))
                dir_map.setdefault(cwd, []).append(int(entry))
            except FILE_ACCESS_ERRORS: continue
    except Exception: pass
    return dir_map
