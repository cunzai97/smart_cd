#!/usr/bin/env python3
"""Smart CD - Scorer module."""

import time
import math
from config import get_scoring_config

class PathScore:
    def __init__(self, path, total, frequency, recency, time_pattern,
                 transition, dwell, visit_count, dwell_time):
        self.path = path
        self.total = total
        self.frequency = frequency
        self.recency = recency
        self.time_pattern = time_pattern
        self.transition = transition
        self.dwell = dwell
        self.visit_count = visit_count
        self.dwell_time = dwell_time

class Scorer:
    def __init__(self):
        cfg = get_scoring_config()
        self.freq_w, self.rec_w, self.time_w = cfg.get('frequency_weight', 0.30), cfg.get('recency_weight', 0.25), cfg.get('time_pattern_weight', 0.10)
        self.trans_w, self.dwell_w = cfg.get('transition_weight', 0.15), cfg.get('dwell_weight', 0.20)
        self.half_life, self.dwell_max = cfg.get('half_life_days', 7), cfg.get('dwell_max_hours', 1.0)

    def score_path(self, visit_count, last_visit, dwell_time, hour_dist, current_hour, max_visits, max_dwell, path, transitions):
        freq = math.log(visit_count + 1) / math.log(max(max_visits, 1) + 1) if visit_count else 0
        recent = 0.5 ** ((time.time() - last_visit) / 86400 / self.half_life)
        time_pat = sum(math.exp(-(min(abs(h-current_hour),24-abs(h-current_hour))**2)/8)*c for h,c in hour_dist) / sum(c for _,c in hour_dist) if hour_dist else 0
        trans = transitions.get(path, 0)
        dwell = min(dwell_time/3600/self.dwell_max, 1)
        return PathScore(path, self.freq_w*freq + self.rec_w*recent + self.time_w*time_pat + self.trans_w*trans + self.dwell_w*dwell,
                         freq, recent, time_pat, trans, dwell, visit_count, dwell_time)

    def rank_paths(self, candidates, transitions, current_hour, max_visits, max_dwell):
        scores = [self.score_path(c.get('visit_count',0), c.get('last_visit',0), c.get('total_dwell_time',0),
                     c.get('hour_distribution',[]), current_hour, max_visits, max_dwell, c['path'], transitions) for c in candidates]
        scores.sort(key=lambda s: s.total, reverse=True)
        return scores
