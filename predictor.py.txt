#!/usr/bin/env python3
"""Smart CD - Predictor module."""

import os
from datetime import datetime
from database import Database
from scorer import Scorer

# 常量
LETTERS = 'abcdefghijklmnopqrstuvwxyz'
MAX_RECOMMENDATIONS = len(LETTERS)
HOME_DIR = os.path.expanduser('~')

class Predictor:
    def __init__(self, db=None):
        self.db = db or Database()
        self.scorer = Scorer()

    def predict(self, current_dir, limit=5):
        """预测推荐目录。"""
        limit = min(limit, MAX_RECOMMENDATIONS)

        # 获取所有需要的数据
        stats = self.db.get_global_stats()
        transitions = self.db.get_transitions_from(current_dir)
        candidates = self.db.get_candidates_for_prediction(limit * 3)
        hour_dists = self.db.get_all_hour_distributions([c['path'] for c in candidates])

        for c in candidates:
            c['hour_distribution'] = hour_dists.get(c['path'], [])

        scores = self.scorer.rank_paths(
            candidates, transitions,
            datetime.now().hour,
            stats.get('max_visits', 1),
            stats.get('max_dwell', 1)
        )

        # 过滤当前目录和 home 目录，验证目录存在性
        result = []
        for s in scores:
            if s.path == current_dir or s.path == HOME_DIR:
                continue
            if not os.path.isdir(s.path):
                continue
            result.append(s)
            if len(result) >= limit:
                break

        return result

    def get_formatted_recommendations(self, current_dir, limit=5):
        """获取格式化的推荐列表。"""
        scores = self.predict(current_dir, limit)
        return [
            {
                'letter': LETTERS[i],
                'path': s.path,
                'visit_count': s.visit_count,
                'dwell_hours': s.dwell_time / 3600,
                'score': s.total,
                'frequency': s.frequency,
                'recency': s.recency,
                'time_pattern': s.time_pattern,
                'transition': s.transition,
                'dwell': s.dwell
            }
            for i, s in enumerate(scores)
        ]

    def get_path_by_letter(self, current_dir, letter, limit=5):
        """根据字母获取推荐路径。"""
        if len(letter) != 1 or not letter.isalpha():
            return None

        idx = ord(letter.lower()) - ord('a')
        if idx < 0 or idx >= MAX_RECOMMENDATIONS:
            return None

        scores = self.predict(current_dir, limit)
        if idx < len(scores):
            return scores[idx].path
        return None

    def close(self):
        self.db.close()

    def __enter__(self):
        return self

    def __exit__(self, *_):
        self.close()
