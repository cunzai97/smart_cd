#!/usr/bin/env python3
"""Smart CD - Fuzzy match module."""

import os
import time
from config import get_fuzzy_config

def fuzzy_match_score(query: str, target: str) -> int:
    """Calculate fuzzy match score. Higher = better."""
    query, target = query.lower(), target.lower()
    qi, ti, score, consecutive, first_match = 0, 0, 0, 0, -1
    while qi < len(query) and ti < len(target):
        if query[qi] == target[ti]:
            if first_match == -1: first_match = ti
            qi += 1; consecutive += 1
            score += consecutive * 10 - ti
        else: consecutive = 0
        ti += 1
    if qi == len(query):
        score += (20 - first_match) * 5 + (30 - len(target)) * 2 + int(len(query)/len(target) * 50)
        return score
    return 0

def find_best_fuzzy_match(query: str, start_dir: str = '.') -> str:
    """Find best fuzzy match respecting config depth and timeout."""
    cfg = get_fuzzy_config()
    max_depth, max_time_ms = cfg.get('max_depth', 3), cfg.get('max_time_ms', 500)
    start_time = time.time()
    best_match, best_score = '', 0
    for entry in os.listdir(start_dir) if start_dir == '.' else []:
        full = os.path.join(start_dir, entry)
        if os.path.isdir(full):
            s = fuzzy_match_score(query, entry)
            if s > best_score: best_score, best_match = s, entry
    for root, dirs, _ in os.walk(start_dir):
        depth = root[len(start_dir):].count(os.sep)
        if depth > max_depth: continue
        if (time.time() - start_time) * 1000 > max_time_ms: break
        for d in dirs:
            s = fuzzy_match_score(query, d) - depth * 5
            if s > best_score:
                best_score = s
                best_match = os.path.join(root, d) if start_dir == '.' else os.path.relpath(os.path.join(root, d), start_dir)
    return best_match
