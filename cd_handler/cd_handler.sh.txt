#!/bin/bash
# cd_handler - CD 处理模块统一入口
# 用法: cd_handler <cd_arg> [--shell=bash|csh] [prev_path]
# 输出: 命令字符串，供 shell eval 执行

D="__DIR__"
DEPTH="__DEPTH__"
SHELL_TYPE="bash"

# 解析参数
ARG=""
P="$PWD"
for param in "$@"; do
    case "$param" in
        --shell=bash) SHELL_TYPE="bash" ;;
        --shell=csh)  SHELL_TYPE="csh" ;;
        --help)       echo 'echo "用法: cd [数字|字母(a-z)|目录|子串|模糊|-]"'; exit 0 ;;
        --*|--)       ;;
        *)
            if [[ -z "$ARG" ]]; then
                ARG="$param"
            else
                P="$param"
            fi
            ;;
    esac
done

# === 输出函数 ===

# 后台执行 cd_hook（非阻塞记录访问）
run_hook() {
    local target="$1"
    local letter="${2:-}"
    ($D/cd_hook "$target" "$P" "$letter" >/dev/null 2>&1 &)
}

output_cd() {
    local target="$1"
    local letter="${2:-}"
    if [[ "$SHELL_TYPE" == "csh" ]]; then
        echo "chdir \"$target\"; setenv LAST_CD_PATH \"$target\"; setenv LAST_CD_TIME \`date +%s\`; $D/ll_num;"
    else
        echo "CD_TARGET=\"$target\"; CD_LETTER=\"$letter\""
    fi
    run_hook "$target" "$letter"
}

output_cd_dash() {
    if [[ "$SHELL_TYPE" == "csh" ]]; then
        echo "chdir -; setenv LAST_CD_PATH \"\$OLDPWD\"; setenv LAST_CD_TIME \`date +%s\`; $D/ll_num;"
    else
        echo "CD_TARGET=\"$OLDPWD\"; CD_DASH=1"
    fi
    run_hook "$OLDPWD"
}

output_cd_home() {
    if [[ "$SHELL_TYPE" == "csh" ]]; then
        echo "chdir; setenv LAST_CD_PATH \"$HOME\"; setenv LAST_CD_TIME \`date +%s\`; $D/ll_num;"
    else
        echo "CD_TARGET=\"$HOME\"; CD_HOME=1"
    fi
    run_hook "$HOME"
}

output_file() {
    local file="$1"
    if [[ "$SHELL_TYPE" == "csh" ]]; then
        echo "gvim \"$file\" &;"
    else
        echo "CD_FILE=\"$file\""
    fi
}

output_error() {
    local msg="$1"
    if [[ "$SHELL_TYPE" == "csh" ]]; then
        echo "echo \"$msg\";"
    else
        echo "CD_ERROR=\"$msg\""
    fi
}

# === 核心逻辑 ===

# cd -
[[ "$ARG" == "-" ]] && { output_cd_dash; exit 0; }

# cd (空) -> home
[[ -z "$ARG" ]] && { output_cd_home; exit 0; }

# 字母推荐 (a-z)
if [[ "$ARG" =~ ^[a-z]$ ]]; then
    letter="$ARG"
    target=$($D/rec_cache_read "$letter" 2>/dev/null)
    if [[ -z "$target" ]] && [[ -f "$D/smart_cd.py" ]]; then
        target=$(python3 "$D/smart_cd.py" --letter "$PWD" "$letter" "$OLDPWD" 2>/dev/null)
    fi
    if [[ -n "$target" ]] && [[ -d "$target" ]]; then
        output_cd "$target" "$letter"
        exit 0
    fi
fi

# 数字导航
if [[ "$ARG" =~ ^[0-9]+$ ]]; then
    t=$(ls -1 2>/dev/null | awk -v n="$ARG" 'NR==n{print;exit}')
    [[ -z "$t" ]] && { output_error "cd: 无条目 #$ARG"; exit 1; }
    [[ -d "$t" ]] && { abs=$(cd "$t" && pwd); output_cd "$abs"; exit 0; }
    [[ -f "$t" ]] && { output_file "$t"; exit 0; }
fi

# 绝对路径
if [[ "$ARG" == /* ]] && [[ -d "$ARG" ]]; then
    output_cd "$ARG"
    exit 0
fi
[[ "$ARG" == /* ]] && [[ -f "$ARG" ]] && { output_file "$ARG"; exit 0; }

# 相对路径目录
if [[ -d "$ARG" ]]; then
    abs=$(cd "$ARG" && pwd)
    output_cd "$abs"
    exit 0
fi
[[ -f "$ARG" ]] && { output_file "$ARG"; exit 0; }

# 子串匹配：选择名称最短的匹配目录（用户输入越短，越倾向精确匹配）
m=$(ls -1 2>/dev/null | awk -v q="$ARG" 'BEGIN{bs=999999;b=""}{if(tolower($0)~tolower(q)){l=length($0);if(l<bs){bs=l;b=$0}}}END{print b}')

# 分层模糊搜索：允许跳字符匹配（如 "scf" 匹配 "smart_cd_faster"）
# 评分公式 sc=s*10+len-cc*5：位置分数(s)为主，名称长度惩罚，连续匹配奖励(cc)
# 排除版本控制/依赖目录：这些目录数量多且用户很少主动进入
if [[ -z "$m" ]] && [[ "$DEPTH" -gt 0 ]]; then
    for d in 1 2 3 4 5; do
        [[ "$d" -gt "$DEPTH" ]] && break
        m=$(find . -maxdepth $d -mindepth $d -type d \
            ! -path '*/.git/*' ! -path '*/.svn/*' ! -path '*/node_modules/*' \
            ! -path '*/__pycache__/*' ! -path '*/.cache/*' ! -path '*/build/*' \
            ! -path '*/dist/*' ! -path '*/.tox/*' ! -path '*/venv/*' \
            ! -name '.git' ! -name '.svn' ! -name 'node_modules' \
            ! -name '__pycache__' ! -name '.cache' ! -name 'build' \
            ! -name 'dist' ! -name '.tox' ! -name 'venv' 2>/dev/null | \
            sed 's|^\./||' | \
            awk -v q="$ARG" 'BEGIN{split(q,c,"");l=length(q);bs=999999}{n=$0;s=0;p=0;cc=0;for(i=1;i<=l;i++){f=index(substr(n,p+1),c[i]);if(f==0)next;np=p+f;if(np==p+1)cc++;s+=f;p=np}sc=s*10+length(n)-cc*5;if(sc<bs){bs=sc;b=n}}END{if(bs<999999)print b}')
        [[ -n "$m" ]] && break
    done
fi

# 找到匹配
if [[ -n "$m" ]]; then
    if [[ -d "$m" ]]; then
        abs=$(cd "$m" && pwd)
        output_cd "$abs"
        exit 0
    fi
    if [[ -f "$m" ]]; then
        output_file "$m"
        exit 0
    fi
fi

# 最后尝试原始 cd
output_error "cd: 无匹配 '$ARG'"
exit 1
