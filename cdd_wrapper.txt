#!/bin/bash
# cdd_wrapper - csh 模糊搜索 cd
# 用法: eval `cdd_wrapper <query>`

query="$1"
script="$HOME/.local/share/smart_cd/smart_cd.py"

# 检查参数
if [[ -z "$query" ]]; then
    echo "echo \"Usage: cdd <query>\""
    exit 1
fi

# 执行模糊搜索
path=$(python3 "$script" --cdd "$query" 2>/dev/null)

# 检查结果
if [[ -z "$path" || "$path" == "cdd:"* ]]; then
    echo "echo \"${path:-cdd: No match found}\""
    exit 0
fi

# 检查是否为目录
if [[ ! -d "$path" ]]; then
    echo "echo \"cdd: Not a directory '$path'\""
    exit 0
fi

# 转义路径中的特殊字符 (csh/tcsh 格式)
escape_path_csh() {
    local p="$1"
    p="${p//\\/\\\\}"
    p="${p//\"/\\\"}"
    p="${p//\$/\\\$}"
    p="${p//\`/\\\`}"
    echo "$p"
}

escaped_path=$(escape_path_csh "$path")

# 生成 chdir 命令（检查 SHOW_RECS）
show_recs="${SMART_CD_SHOW_RECS:-1}"
recs_first="${SMART_CD_RECS_FIRST:-0}"

cmd="chdir \"$escaped_path\" ; python3 \"$script\" --record \"\$cwd\""

if [[ "$show_recs" == "0" ]]; then
    cmd="$cmd ; ~/.local/share/smart_cd/ll_num"
else
    if [[ "$recs_first" == "1" ]]; then
        cmd="$cmd ; python3 \"$script\" --ll ; ~/.local/share/smart_cd/ll_num"
    else
        cmd="$cmd ; ~/.local/share/smart_cd/ll_num ; python3 \"$script\" --ll"
    fi
fi

echo "$cmd"
