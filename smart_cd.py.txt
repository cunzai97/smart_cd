#!/usr/bin/env python3
"""Smart CD - Main CLI entry point."""

import os
import sys
import shlex
import argparse
import subprocess
from datetime import datetime

from config import get_display_config, get_editor_command, get_custom_commands, HOME_DIR
from database import Database
from tracker import Tracker
from predictor import Predictor
from daemon import daemon_main
from display import format_recommendations
from fuzzy import find_best_fuzzy_match

HELP_TEXT = """
Smart CD - 智能目录导航工具
============================

快捷命令:
  sl          显示推荐列表（不切换目录）
  sr          切换推荐显示开关
  so          切换输出顺序(目录先/推荐先)
  sc          切换底部对齐输出
  cdd <query> 模糊匹配cd（允许跳字符）
  cd --help   显示此帮助

基本用法:
  cd          回到home目录并显示推荐
  cd <数字>   进入第N个目录/文件 (基于ll编号)
  cd <字母>   进入推荐的目录 (a=第1推荐, b=第2推荐...)
  cd <目录>   进入匹配的目录 (当前目录子串匹配)

配置文件:
  ~/.local/share/smart_cd/config.ini
"""

# 常量
PREV_DIR_FLAG = '-'  # 返回上一次目录的标志
HOME_FLAG = '~'       # 返回 home 的标志
MAX_NUMERIC_ARG = 100000  # 最大数字参数限制

def get_directory_entries(include_hidden=False):
    """获取当前目录条目列表，保持 ls 的原始顺序。"""
    try:
        output = subprocess.check_output(['/bin/ls', '-1'], stderr=subprocess.DEVNULL)
        entries = output.decode('utf-8').strip().split('\n')
        if not include_hidden:
            entries = [e for e in entries if e and not e.startswith('.')]
        return entries
    except subprocess.CalledProcessError:
        # 回退到 os.listdir
        entries = os.listdir('.')
        if not include_hidden:
            entries = [e for e in entries if not e.startswith('.')]
        return sorted(entries)
    except Exception:
        return []

def match_substring(query):
    """子串匹配，返回最短的匹配项。"""
    try:
        entries = os.listdir('.')
        matches = [e for e in entries if query.lower() in e.lower()]
        return min(matches, key=len) if matches else None
    except OSError:
        return None

def escape_path(path):
    """转义路径中的特殊字符。"""
    return shlex.quote(path)

def main():
    parser = argparse.ArgumentParser(add_help=False)
    parser.add_argument('--help', '-h', action='store_true')
    parser.add_argument('--init', action='store_true')
    parser.add_argument('--record', metavar='PATH')
    parser.add_argument('--prev', metavar='PATH')
    parser.add_argument('--predict', '--ll', action='store_true')
    parser.add_argument('--get', metavar='LETTER')
    parser.add_argument('--resolve', metavar='ARG')
    parser.add_argument('--toggle', metavar='VAR')  # 切换环境变量
    parser.add_argument('--cdd', metavar='QUERY')
    parser.add_argument('--daemon', metavar='ACTION', choices=['start', 'stop', 'status'])
    parser.add_argument('--stats', action='store_true')
    parser.add_argument('--cleanup', metavar='DAYS', type=int)
    parser.add_argument('--limit', type=int, default=get_display_config().get('recommendation_limit', 5))
    args = parser.parse_args()

    if args.help:
        print(HELP_TEXT)
        return

    if args.init:
        Database()
        print("smart_cd database initialized")
        return

    if args.record:
        Tracker().record_visit(args.record, args.prev)
        return

    if args.daemon:
        daemon_main(args.daemon)
        return

    # 解析参数
    if args.resolve is not None:
        arg = args.resolve.strip() if args.resolve else ''

        # 帮助
        if arg in ('--help', '-h'):
            print(HELP_TEXT)
            return

        # 空参数 -> home
        if arg == '':
            print(HOME_FLAG)
            return

        # cd - 返回上一次目录
        if arg == PREV_DIR_FLAG:
            print(PREV_DIR_FLAG)
            return

        # 字母 -> 推荐
        if len(arg) == 1 and arg.isalpha() and arg.islower():
            path = Predictor().get_path_by_letter(os.getcwd(), arg, args.limit)
            print(path if path else f'cd: No recommendation for letter \'{arg}\'')
            return

        # 数字 -> 条目
        if arg.isdigit():
            num = int(arg)
            # 边界检查
            if num <= 0 or num > MAX_NUMERIC_ARG:
                print(f'cd: Invalid entry number #{arg}')
                return

            entries = get_directory_entries()
            if 0 < num <= len(entries):
                print(entries[num - 1])
            else:
                print(f'cd: No entry #{arg}')
            return

        # 存在的路径
        if os.path.isdir(arg) or os.path.isfile(arg):
            print(arg)
            return

        # 子串匹配
        matched = match_substring(arg)
        print(matched if matched else f'cd: No match for \'{arg}\'')
        return

    # 显示推荐
    if args.predict:
        recs = Predictor().get_formatted_recommendations(os.getcwd(), args.limit)
        print(format_recommendations(recs))
        return

    # 切换环境变量
    if args.toggle:
        current = os.environ.get(args.toggle, '0')
        print('0' if current == '1' else '1')
        return

    # 获取推荐路径
    if args.get:
        path = Predictor().get_path_by_letter(os.getcwd(), args.get, args.limit)
        if path:
            print(path)
        else:
            sys.exit(1)
        return

    # 模糊搜索
    if args.cdd:
        match = find_best_fuzzy_match(args.cdd, '.')
        print(match if match else f"cdd: No fuzzy match for '{args.cdd}'")
        return

    # 统计信息
    if args.stats:
        stats = Database().get_global_stats()
        print(f"Paths: {stats.get('total_paths',0)}, Max visits: {stats.get('max_visits',0)}, Max dwell: {stats.get('max_dwell',0)/3600:.1f}h")
        return

    # 清理旧记录
    if args.cleanup:
        Database().cleanup_old_records(args.cleanup)
        print(f"Cleaned records older than {args.cleanup} days")
        return

    parser.print_help()

def cdd_main():
    """cdd 命令入口。"""
    if len(sys.argv) < 2:
        print("Usage: cdd <query> [start_dir]")
        sys.exit(1)

    query = sys.argv[1]
    start_dir = sys.argv[2] if len(sys.argv) > 2 else '.'
    match = find_best_fuzzy_match(query, start_dir)

    if match:
        print(match)
        sys.exit(0)
    else:
        print(f"cdd: No fuzzy match for '{query}'")
        sys.exit(1)

if __name__ == '__main__':
    # 检测是否作为 cdd 运行
    if os.path.basename(sys.argv[0]) == 'cdd':
        cdd_main()
    elif len(sys.argv) > 1 and sys.argv[1] == '--cdd':
        sys.argv = sys.argv[:1] + sys.argv[2:]
        cdd_main()
    else:
        main()
